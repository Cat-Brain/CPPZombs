#version 330 core
layout (location = 0) out vec4 FragColor;
layout (location = 1) out vec4 normal;

uniform vec4 color;

in vec2 pos;

void main()
{
	// Don't draw if you're outside of the bounds of the circle, this is done as the mesh will always overdraw.
	if (dot(pos, pos) > 1) discard;
	FragColor = color;

	// Normal calculation is sort of weird and probably not accurate:
	normal = vec4(pos.x, pos.y, sqrt(1. - pos.x * pos.x - pos.y * pos.y), color.a);
}
