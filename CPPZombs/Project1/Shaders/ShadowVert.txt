#version 330 core
layout (location = 0) in vec2 aPos;

uniform vec2 scale, position;
uniform vec2 bottomLeft, topRight;
out vec2 pos, screenPos;

void main()
{
	screenPos = vec2(aPos.x * scale.x + position.x, aPos.y * scale.y + position.y);
	gl_Position = vec4(screenPos, 0., 1.);
	pos = bottomLeft * (1 - aPos) + topRight * aPos;
}
