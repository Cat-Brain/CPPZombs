#version 330 core
out vec4 FragColor;

in vec2 screenPos;
uniform sampler2D normalMap;
uniform vec3 color, direction;
uniform bool colorBand;

void main()
{
	vec4 normal = texture(normalMap, screenPos * 0.5 + 0.5);
	float diff = dot(normal.xyz, -direction) * 0.5 + 0.5;
	if (colorBand) diff = 0.2 * round(diff * 5); // Round for color banding.
	FragColor = vec4(color * diff, 1.);
}
